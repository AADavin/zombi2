% Extra preamble for the TOOLS PDF only (appended after preamble.tex).
%
% The docs/tools/ pages are authored for MkDocs, where the browser's fonts cover
% every glyph, so they use literal Unicode math symbols (τ, ≤, ℓ, subscripts, …).
% The main manual writes these as LaTeX math ($\tau$) and never hits the gap, but
% the generated tools chapters carry the literal characters — and Latin Modern has
% no glyph for them. Map each to its LaTeX equivalent so XeLaTeX renders it instead
% of dropping it. Kept out of the shared preamble.tex so the main book is untouched.
% Wrap long code lines instead of letting them run off the right margin. The
% tools pages carry wide CLI examples and example-output blocks (authored for a
% browser, which just scrolls); in print they must break. fvextra adds line
% breaking to the fancyvrb Verbatim that pandoc's syntax highlighting uses.
\usepackage{fvextra}
\fvset{breaklines=true,breakanywhere=true}

\usepackage{newunicodechar}
\newunicodechar{δ}{\ensuremath{\delta}}
\newunicodechar{τ}{\ensuremath{\tau}}
\newunicodechar{λ}{\ensuremath{\lambda}}
\newunicodechar{ℓ}{\ensuremath{\ell}}
\newunicodechar{≤}{\ensuremath{\leq}}
\newunicodechar{≥}{\ensuremath{\geq}}
\newunicodechar{⇒}{\ensuremath{\Rightarrow}}
\newunicodechar{√}{\ensuremath{\surd}}
\newunicodechar{₁}{\textsubscript{1}}
\newunicodechar{₂}{\textsubscript{2}}
\newunicodechar{⁴}{\textsuperscript{4}}
